use displaydoc::Display;
use thiserror::Error;

/// factory result
pub type FactoryResult<T, E = FactoryError> = core::result::Result<T, E>;

/// factory error
#[non_exhaustive]
#[derive(Display, Error, Debug)]

    /// Generic error: {0}

}
